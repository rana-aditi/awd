<script>
function fact(){
var i, n, factorial;
factorial= 1;
n = document.getElementById("n").value;
for(i = 1; i <= n; i++)
{
factorial = factorial * i;
}
i = i - 1;
document.getElementById("rev").innerHTML = "The factorial of the number " + i + " is: " + factorial ;
}
</script>
